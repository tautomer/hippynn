    Molecules   = "Molecules"
    Atoms       = "Atoms"
    Pair        = "Pair"

    QuadMol     = "QuadMol"
    QuadPack    = "QuadPack"      # packed 6-vec of quadrupole upper triangle
    NotFound    = "NOT FOUND"
