    def forward(self, pair_dist, *pair_tensors):
        r_cut = self.hard_dist_cutoff 
        idx = torch.argwhere(pair_dist <= r_cut).squeeze(1)
        pair_tensors = pair_dist, *pair_tensors
        return tuple(pl[idx] for pl in pair_tensors)
